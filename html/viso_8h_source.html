32  // camera parameters (all are mandatory / need to be supplied)

48  double bucket_height; // height of bucket
49  bucketing () {
50  max_features = 2;

63  // constructor, takes as input a parameter structure:

71  // call this function instead of the specialized ones, if you already have
72  // feature matches, and simply want to compute visual odometry from them, without
73  // using the internal matching functions.
74  bool process (std::vector<Matcher::p_match> p_matched_) {
75  p_matched = p_matched_;
76  return updateMotion();
77  }
78 
79  // returns transformation from previous to current coordinates as a 4x4
80  // homogeneous transformation matrix Tr_delta, with the following semantics:
81  // p_t = Tr_delta * p_ {t-1} takes a point in the camera coordinate system
82  // at time t_1 and maps it to the camera coordinate system at time t.
83  // note: getMotion() returns the last transformation even when process()
84  // has failed. this is useful if you wish to linearly extrapolate occasional
85  // frames for which no correspondences have been found
86  Matrix getMotion () { return Tr_delta; }
87 
88  // returns previous to current feature matches from internal matcher
89  std::vector<Matcher::p_match> getMatches () { return matcher->getMatches(); }
90 
91  // returns the number of successfully matched points, after bucketing
92  int32_t getNumberOfMatches () { return p_matched.size(); }
93 
94  // returns the number of inliers: num_inliers <= num_matched
95  int32_t getNumberOfInliers () { return inliers.size(); }
96 
97  // returns the indices of all inliers
98  std::vector<int32_t> getInlierIndices () { return inliers; }
99 
100  // given a vector of inliers computes gain factor between the current and
101  // the previous frame. this function is useful if you want to reconstruct 3d
102  // and you want to cancel the change of (unknown) camera gain.
103  float getGain (std::vector<int32_t> inliers_) { return matcher->getGain(inliers_); }
104 
105  // streams out the current transformation matrix Tr_delta
106  friend std::ostream& operator<< (std::ostream &os,VisualOdometry &viso) {
107  Matrix p = viso.getMotion();
108  os << p.val[0][0] << " " << p.val[0][1] << " " << p.val[0][2] << " " << p.val[0][3] << " ";
109  os << p.val[1][0] << " " << p.val[1][1] << " " << p.val[1][2] << " " << p.val[1][3] << " ";
110  os << p.val[2][0] << " " << p.val[2][1] << " " << p.val[2][2] << " " << p.val[2][3];
111  return os;
112  }
